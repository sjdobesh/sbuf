68 /**
69  * allocates a new buffer.
70  * note that allocation is actually 1 larger than specified.
71  * buf[capacity + 1] is always string terminator to prevent strfunction errors.
72  *
73  * @param string the data to store in the buffer
74  * @param capacity the max length of string the buffer can store
75  * @return a newly allocated sbuf struct, empty on failure.
76  */
77 sbuf new_sbuf(char* string, size_t capacity) {
78  sbuf s;
79  int len;
80  s.buf = malloc(capacity + 1); /* add room for '\0' */
81  if (s.buf == NULL) {
82  errno = ENOMEM;
83  } else {
84  s.capacity = capacity;
85  len = strnlen(string, capacity);
86  strncpy(s.buf, string, len);
87  s.buf[capacity] = '\0';
88  s.len = len;
89  }
90  return s;
91 }
