70 /**
71  * allocates a new buffer of a specific size.
72  * note that allocation is actually 1 larger than specified.
73  * buf[capacity + 1] is always string terminator to prevent strfunction errors.
74  *
75  * @param string the data to store in the buffer
76  * @param capacity the max length of string the buffer can store
77  * @return a newly allocated sbuf struct, empty on failure.
78  */
79 sbuf new_sbuf_size(char* string, size_t capacity) {
80  sbuf s;
81  int len;
82  errno = 0;
83  s.buf = malloc(capacity + 1); /* add room for '\0' */
84  if (s.buf == NULL) {
85  errno = ENOMEM;
86  } else {
87  s.capacity = capacity;
88  len = strnlen(string, capacity);
89  strncpy(s.buf, string, len);
90  s.buf[capacity] = '\0';
91  s.len = len;
92  }
93  return s;
94 }

165 /**
166  * appends a string to a sbuf.
167  *
168  * @param string the content to append
169  * @param s the sbuf to append to
170  * @return an exit code, setting errno on failure
171  */
172 int sbuf_append_str(sbuf* s, char* string) {
173  int space, len;
174  errno = 0;
175  if (s->buf == NULL) {
176  errno = EPERM;
177  return 1;
178  }
179  /* check length requirements. if dynamic check beyond limit */
180  space = s->capacity - s->len;
181  len = s->dynamic
182  ? strlen(string)
183  : strnlen(string, space);
184  if (len > space) {
185  /* if we are out of space, check if we can resize */
186  if (s->dynamic) {
187  sbuf_realloc(s, s->len + len - space);
188  } else {
189  errno = ENOMEM;
190  return 1;
191  }
192  }
193  strncat(s->buf, string, len);
194  s->len += len;
195  return 0;
196 }
